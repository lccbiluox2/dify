package com.dify.api.utils;

import com.fasterxml.jackson.core.JsonProcessingException;
import com.fasterxml.jackson.databind.ObjectMapper;
import com.fasterxml.jackson.databind.SerializationFeature;

import java.nio.charset.StandardCharsets;
import java.security.MessageDigest;
import java.security.NoSuchAlgorithmException;
import java.util.Map;

/**
 * Dify工作流哈希工具类
 * 算法：SHA256(JSON.stringify({"graph": graph_dict}, sort_keys=True) + "None")
 */
public class WorkflowHashUtil {

    private static final ObjectMapper objectMapper = new ObjectMapper()
            .configure(SerializationFeature.ORDER_MAP_ENTRIES_BY_KEYS, true);

    /**
     * 生成工作流的唯一哈希值
     * 算法：SHA256(JSON.stringify({"graph": graph}) + "None")
     *
     * @param graph 工作流的图形定义（节点和边）
     * @return 哈希值（64位十六进制字符串）
     */
    public static String generateWorkflowHash(Map<String, Object> graph) {
        try {
            // 1. 构建 entity 对象
            Map<String, Object> entity = Map.of("graph", graph);

            // 2. 使用 Jackson 将对象序列化为 JSON 字符串（键按字母排序）
            String jsonString = objectMapper.writeValueAsString(entity);

            // 3. 拼接 "None"
            String hashInput = jsonString + "None";

            // 4. 计算 SHA-256 哈希
            return sha256(hashInput);

        } catch (JsonProcessingException e) {
            throw new RuntimeException("Failed to generate workflow hash", e);
        }
    }

    /**
     * 直接从JSON字符串生成哈希（用于测试场景）
     *
     * @param rawJson 完整的请求JSON字符串
     * @return 哈希值（64位十六进制字符串）
     */
    public static String generateWorkflowHashFromJson(String rawJson) {
        try {
            // 1. 解析JSON获取graph对象
            com.fasterxml.jackson.databind.ObjectMapper mapper = new com.fasterxml.jackson.databind.ObjectMapper();
            @SuppressWarnings("unchecked")
            Map<String, Object> requestPayload = mapper.readValue(rawJson, Map.class);
            
            // 2. 提取graph
            @SuppressWarnings("unchecked")
            Map<String, Object> graph = (Map<String, Object>) requestPayload.get("graph");
            
            if (graph == null) {
                throw new IllegalArgumentException("JSON中未找到graph字段");
            }
            
            // 3. 调用主方法生成哈希
            return generateWorkflowHash(graph);
            
        } catch (Exception e) {
            throw new RuntimeException("Failed to generate workflow hash from JSON", e);
        }
    }

    /**
     * 计算 SHA-256 哈希
     */
    private static String sha256(String input) {
        try {
            MessageDigest digest = MessageDigest.getInstance("SHA-256");
            byte[] hash = digest.digest(input.getBytes(StandardCharsets.UTF_8));
            return bytesToHex(hash);
        } catch (NoSuchAlgorithmException e) {
            throw new RuntimeException("SHA-256 algorithm not found", e);
        }
    }

    /**
     * 将字节数组转换为十六进制字符串
     */
    private static String bytesToHex(byte[] bytes) {
        StringBuilder hexString = new StringBuilder();
        for (byte b : bytes) {
            String hex = Integer.toHexString(0xff & b);
            if (hex.length() == 1) {
                hexString.append('0');
            }
            hexString.append(hex);
        }
        return hexString.toString();
    }
}
